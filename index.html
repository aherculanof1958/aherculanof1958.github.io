function renderizarLinha(container, tipo) { container.innerHTML = ''; for (let i = 0; i < TOTAL_ANOS; i++) { const cel = document.createElement('div'); cel.className = `celula ${tipo}`; const ano = ANO_INICIAL + i; // Filtra todos os eventos deste ano e categoria const evs = eventos[tipo].filter(e => e["ano civil"] == ano); if (evs.length > 0) { cel.classList.add('com-evento'); // Constrói o texto da etiqueta (Tooltip) // Começa com o ano em destaque let texto = `ANO ${ano}\n----------------\n`; // Adiciona cada título de evento evs.forEach(ev => { texto += `• ${ev.titulo}\n`; }); // Atribui ao title da célula cel.title = texto.trim(); } container.appendChild(cel); } }
